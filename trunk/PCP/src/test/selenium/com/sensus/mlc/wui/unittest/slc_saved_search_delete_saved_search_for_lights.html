delete_saved_search_for_lights
click id=saved
waitForNotVisible id=loading
storeTableRow ${searchName}/1 row
storeEval new Number(storedVars['row']) - 2 row
echo ${row}
click css=#saved-table tbody tr:eq(${row}) td:eq(7) a
waitForNotVisible id=loading
waitForVisible id=selected "${searchName}" will be permanently deleted.
assertText css=#action-dialog-delete > #selected > span “${searchName}“ will be permanently deleted.
click //button[@type='button']
waitForNotVisible id=loading
waitForText css=#messaging-main > span Saved Search was successfully deleted.
assertText css=#messaging-main > span Saved Search was successfully deleted.
storeTableRow ${searchName}-Changed/1 row
storeEval new Number(storedVars['row']) - 2 row
echo ${row}
click css=#saved-table tbody tr:eq(${row}) td:eq(7) a
waitForNotVisible id=loading
waitForVisible id=action-dialog "${searchName}-Changed" will be permanently deleted.
click //button[@type='button']
waitForNotVisible id=loading
waitForText css=#messaging-main > span Saved Search was successfully deleted.
assertText css=#messaging-main > span Saved Search was successfully deleted.
